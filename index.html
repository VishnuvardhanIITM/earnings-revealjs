Net Income Sensitivity (Python)


import pandas as pd

base_net_income = 620_000_000

def simulate(rate):
    spread = 1 + (rate - 3.5) * 0.03
    return round(base_net_income * spread, 2)

for r in [3.5, 4.0, 4.5]:
    print(f"Rate {r}% → Net Income: ${simulate(r):,.0f}")
